\newenvironment{pseudocode}[1]{
    \vspace{-0.1in}
    \begin{codebox}
    \noindent
    {\bfseries function} {#1}

    \begin{algorithmic}[1]
    }
    {
    \end{algorithmic}
    \end{codebox}
    }

\newenvironment{pseudocodeT}[1]{
    \vspace{-0.1in}
    \begin{codebox}[sharp corners=south,bottomrule=0pt]
    \noindent
    {\bfseries function} {#1}

    \begin{algorithmic}[1]
    }
    {
    \algstore{pseudocode}
    \end{algorithmic}
    \end{codebox}
    }

\newenvironment{pseudocodeM}{
    \vspace{-0.1in}
    \begin{codebox}[sharp corners=all,bottomrule=0pt,toprule=0pt]
    \begin{algorithmic}[1]
    \algrestore{pseudocode}
    }
    {
    \algstore{pseudocode}
    \end{algorithmic}
    \end{codebox}
    }

\newenvironment{pseudocodeB}{

    \end{algorithmic}
    \end{codebox}
    }
